### 延时函数与读取一个按键的函数

```c
#include <reg51.h>
sbit P10 = P1^0;

/*-----------------------------------------------------
*@Name: delay
*@Function: 实现软件延时
*@Time: 2018-06-15
------------------------------------------------------*/
void delay()
{
    int i,j;
    for(i = 0; i < 110; i++)
    {
        j = 1000;
        while(j-- < 0);
    }
}
/*----------------------------------------------------
*@Name: readKey
*@Function: 读取P10口的状态，为1时返回1.
*@Time: 2018-06-15
----------------------------------------------------*/
int readKey()
{
    if(P10 == 0)
    {
        delay();
        if(0 == P10)
        {
            while(0 == P10);
            return 1;
        }
    }
    return 0;
}
```
